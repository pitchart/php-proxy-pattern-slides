class Customer {
    public function __construct(DbConnection $db, $id) {
        $this->db = $db;
        $this->id = $id;
    }

    public function getName() {
        $this->init();
        return $this->data['name'];
    }

    private function init() {
        if (!isset($this->data)) {
            $this->data = $this->db->load($this->id);
        }
    }
}
