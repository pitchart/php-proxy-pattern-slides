class ImageVirtualProxy extends Image {
    protected $wrapped;
    public function __construct($path) {
        $this->path = $path;
    }

    public function getSize() {
        $this->init();
        return $this->wrapped->getSize();
    }

    private function init() {
        if ( ! $this->wrapped) {
            $this->wrapped = new Image($this->path);
        }
    }
}
